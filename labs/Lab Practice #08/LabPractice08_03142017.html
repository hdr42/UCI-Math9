# Using the closed form solution from lecture.
# N is total number of sample points we have. This was defined above.
S1 = np.sum(X)
S2 = np.sum(X * X)  # sum of the squares
SM = np.sum(X * Z)   # sum of x_i * y_i
SZ = np.sum(Z)
a = (1 / (N*S2 - S1*S1)) * (N * SM - S1 * SZ)
b = (1 / (N*S2 - S1*S1)) * (- S1 * SM + S2 * SZ)

plt.scatter(X, Z, alpha=0.3)
XX = np.linspace(0,5,200).reshape((-1,1))
ZZ = a * XX + b
plt.scatter(X, Z, alpha=0.2)
plt.plot(XX,ZZ,"r")
print("Approximation:\n","a = ",a,"\n b = ",b)
plt.show()
