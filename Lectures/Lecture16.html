# Let's make our lives easier with a helper function
def showimg(imvec):
    plt.figure(figsize=(2,2))
    img = np.reshape(imvec,(24,24)) # convert vectorized data point t 
    plt.imshow( img.T , cmap="gray", interpolation="none")

    showimg(X[i,:])
